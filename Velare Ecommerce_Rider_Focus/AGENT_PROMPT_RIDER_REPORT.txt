4. SUPABASE QUERIES

   Get Deliveries:
   ```dart
   final response = await _supabase
     .from('deliveries')
     .select('''
       delivery_id,
       order_id,
       orders!inner(
         order_id,
         order_number,
         buyer_id,
         seller_id,
         buyers!inner(
           buyer_id,
           first_name,
           last_name,
           user_id
         ),
         sellers!inner(
           seller_id,
           shop_name,
           user_id
         )
       )
     ''')
     .eq('rider_id', riderId)
     .order('created_at', ascending: false);
   ```

   Submit Report:
   ```dart
   // 1. Get reported_user_id
   if (reportedType == 'buyer') {
     final response = await _supabase
       .from('buyers')
       .select('user_id')
       .eq('buyer_id', reportedId)
       .single();
     reportedUserId = response['user_id'];
   } else {
     final response = await _supabase
       .from('sellers')
       .select('user_id')
       .eq('seller_id', reportedId)
       .single();
     reportedUserId = response['user_id'];
   }

   // 2. Upload evidence if provided
   if (evidenceFile != null) {
     final fileName = 'report_${DateTime.now().millisecondsSinceEpoch}_${path.basename(evidenceFile.path)}';
     await _supabase.storage
       .from('reports')
       .upload(fileName, evidenceFile);
     evidencePath = _supabase.storage.from('reports').getPublicUrl(fileName);
   }

   // 3. Insert report
   await _supabase.from('user_reports').insert({
     'reporter_id': currentUserId,
     'reporter_type': 'rider',
     'reported_user_id': reportedUserId,
     'reported_user_type': reportedType,
     'report_category': category,
     'report_reason': reason,
     'order_id': orderId,
     'delivery_id': deliveryId,
     'evidence_image': evidencePath,
     'status': 'pending',
   });

   // 4. Update report count
   final countResponse = await _supabase
     .from('user_reports')
     .select('report_id', count: CountOption.exact)
     .eq('reported_user_id', reportedUserId);

   if (reportedType == 'buyer') {
     await _supabase
       .from('buyers')
       .update({'report_count': countResponse.count})
       .eq('user_id', reportedUserId);
   } else {
     await _supabase
       .from('sellers')
       .update({'report_count': countResponse.count})
       .eq('user_id', reportedUserId);
   }
   ```
